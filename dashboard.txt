<script>

const gpsData = {{ data.report.gps_path|tojson if data else '[]' }};

let map;

if(gpsData.length > 0){

    map = L.map('map').setView(gpsData[0],16);

    L.tileLayer(
        'https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png',
        {
            attribution:'OpenStreetMap'
        }
    ).addTo(map);

    const polyline = L.polyline(
        gpsData,
        {
            color:'red',
            weight:4
        }
    ).addTo(map);

    map.fitBounds(polyline.getBounds());

    L.marker(gpsData[0])
        .addTo(map)
        .bindPopup("Takeoff");

    L.marker(gpsData[gpsData.length - 1])
        .addTo(map)
        .bindPopup("Landing");
}

let replayMarker;
let idx = 0;

function startReplay(){

    if(gpsData.length === 0) return;

    idx = 0;

    if(replayMarker){
        map.removeLayer(replayMarker);
    }

    replayMarker = L.marker(
        gpsData[0]
    ).addTo(map);

    const interval = setInterval(()=>{

        replayMarker.setLatLng(
            gpsData[idx]
        );

        idx++;

        if(idx >= gpsData.length){
            clearInterval(interval);
        }

    },200);

}

</script>
